# Normalno
shapiro_slaba = stats.shapiro(grupa_slaba)
shapiro_dobra = stats.shapiro(grupa_dobra)

print(f"Test Shapiro-Wilka (Sabe wina): p-value = {shapiro_slaba.pvalue}")
print(f"Test Shapiro-Wilka (Dobre wina): p-value = {shapiro_dobra.pvalue}")
if shapiro_slaba.pvalue < 0.05 or shapiro_dobra.pvalue < 0.05:
    print("-> Wniosek: Rozkady nie s normalne (ale ze wzgldu na N > 30 stosujemy test t na podst. Centralnego Twierdzenia Granicznego).")
else:
    print("-> Wniosek: Rozkady s normalne.")

# B. Jednorodno wariancji (Test Levene'a)
# To decyduje, czy uyjemy klasycznego testu t, czy testu Welcha (z poprawk na wariancj).
levene_test = stats.levene(grupa_slaba, grupa_dobra)
print(f"Test Levene'a: p-value = {levene_test.pvalue}")

rowne_wariancje = True
if levene_test.pvalue < 0.05:
    print("-> Wniosek: Wariancje S RNE (uyjemy parametru equal_var=False).")
    rowne_wariancje = False
else:
    print("-> Wniosek: Wariancje s jednorodne.")

# 3. WYKONANIE TESTU T-STUDENTA
print("\n[2.1.3 Wykonanie Testu t-Studenta]")

# alternative='less' oznacza H1: rednia grupy pierwszej (sabe) jest MNIEJSZA ni drugiej (dobre)
t_stat, p_val = stats.ttest_ind(grupa_slaba, grupa_dobra, equal_var=rowne_wariancje, alternative='less')

def kategoryzuj_jakosc(score):
    if score <= 4:
        return 'Niska (3-4)'
    elif score <= 6:
        return 'rednia (5-6)'
    else:
        return 'Wysoka (7-8)'

# Klasyfikacja na grupy bazujc na ocenach oraz utworzenie podzielonych zbiorw obecnoci siarczanw
df['Grupa_Jakoci'] = df['Ocena'].apply(kategoryzuj_jakosc)
zmienna_anova = 'Siarczany'

grupa_niska = df[df['Grupa_Jakoci'] == 'Niska (3-4)'][zmienna_anova]
grupa_srednia = df[df['Grupa_Jakoci'] == 'rednia (5-6)'][zmienna_anova]
grupa_wysoka = df[df['Grupa_Jakoci'] == 'Wysoka (7-8)'][zmienna_anova]

print(f"Liczebnoci grup: \n{df['Grupa_Jakoci'].value_counts()}")

# === KROK 2: Sprawdzenie zaoe (2.2.2) ===
print("\n--- 2.2.2 Wymagania testu ---")
# Normalno (Shapiro) - formalno, bo przy N>30 dziaa CTG
print(f"Shapiro (Niska): p={stats.shapiro(grupa_niska).pvalue}")
print(f"Shapiro (rednia): p={stats.shapiro(grupa_srednia).pvalue}")
print(f"Shapiro (Wysoka): p={stats.shapiro(grupa_wysoka).pvalue}")

# Jednorodno wariancji (Levene)
stat_lev, p_lev = stats.levene(grupa_niska, grupa_srednia, grupa_wysoka)
print(f"Test Levene'a: p={p_lev}")
if p_lev < 0.05:
    print("Uwaga: Wariancje nie s rwne.")

# === KROK 3: Wykonanie ANOVA (2.2.3) ===
print("\n--- 2.2.3 Wykonanie ANOVA ---")
f_stat, p_val = stats.f_oneway(grupa_niska, grupa_srednia, grupa_wysoka)

print(f"Statystyka F: {f_stat}")
print(f"P-value: {p_val:}")

if p_val < 0.05:
    print("\nWNIOSEK: Odrzucamy H0. Istniej istotne rnice midzy grupami.")
    print("Przeprowadzamy test post-hoc Tukeya, aby sprawdzi, ktre grupy si rni.")
    tukey = pairwise_tukeyhsd(endog=df[zmienna_anova], groups=df['Grupa_Jakoci'], alpha=0.05)
    print(tukey)
else:
    print("\nWNIOSEK: Brak podstaw do odrzucenia H0. rednie s rwne.")

print("=== 3. TESTY NIEPARAMETRYCZNE ===")

# --- 3.1 Test Manna-Whitneya (Wilcoxona dla sumy rang) ---
zmienna_mw = 'Chlorki'

grupa_1 = df[df['Klasa'] == 1][zmienna_mw]

print(f"\n[3.1] Test Manna-Whitneya dla zmiennej: {zmienna_mw}")
# Sprawdzamy normalno, eby uzasadni wybr testu

print(f"Shapiro (Dobre): p={stats.shapiro(grupa_1).pvalue}")

u_stat, p_val_mw = stats.mannwhitneyu(grupa_0, grupa_1, alternative='two-sided')
print(f"Wynik U-testu: p-value = {p_val_mw:.20f}")

if p_val_mw < 0.05:
    print("-> Rnica istotna statystycznie.")
else:
    print("-> Brak rnic.")
